Optional

Создание

Optional<String> fullOpt = Optional.of("Hello"); //of бесполезен так как не уберегает от NPE, нужно использовать ofNullable
Optional<String> emptOpt = Optional.empty();

System.out.println(fullOpt.isPresent()); // true
System.out.println(emptOpt.isEmpty()); // false

orElse
System.out.println(fullOpt.orElse("world")); // Hello
System.out.println(emptOpt.orElse("world")); // world

ofNullable - предназначен как раз для потенциально null объектов
// Optional<String> emptOpt2 = Optional.of(null); // NPE т.к. of не предназначен для null
Optional<String> emptOpt2 = Optional.ofNullable(null);
System.out.println(emptOpt2.orElse("world")); // world

map - изменят наш объект даже меня тип возвращаемого значение
System.out.println(fullOpt.map(String::toUpperCase).orElse("WORLD"));
System.out.println(emptOpt2.map(String::toUpperCase).orElse("WORLD"));

ifPresent - ничего не позвращает, просто че то делает если optioanl не пустой
fullOpt.ifPresent(e -> System.out.println("ifPresent " + e));
// для пустого emptOpt просто не сработает

filter по сути делает валидацию на соответвие заданному условию
System.out.println(fullOpt.filter(o -> o.equals("jello")).orElse("no"));
